4.3. Minimum C Code

As a GObject, the extension requires a minimum level of code, which is mostly copy-paste work. Place the following in ephy-foo-extension.c:

#ifdef HAVE_CONFIG_H
#include "config.h"
#endif

#include "ephy-foo-extension.h"
#include "ephy-debug.h"

#include <epiphany/ephy-extension.h>

#include <gmodule.h>

#define EPHY_FOO_EXTENSION_GET_PRIVATE(object) (G_TYPE_INSTANCE_GET_PRIVATE ((object), EPHY_TYPE_FOO_EXTENSION, EphyFooExtensionPrivate))

struct EphyFooExtensionPrivate
{
};

static void ephy_foo_extension_class_init (EphyFooExtensionClass *klass);
static void ephy_foo_extension_iface_init (EphyExtensionClass *iface);
static void ephy_foo_extension_init (EphyFooExtension *extension);

enum
{
  PROP_0
  /* Extension-specific properties go here, to be added using
   * g_object_class_install_property in the
   * class_init function.
   */
};

static GObjectClass *foo_extension_parent_class = NULL;

static GType foo_extension_type = 0;

GType
ephy_foo_extension_get_type (void)
{
  return foo_extension_type;
}

GType
ephy_foo_extension_register_type (GTypeModule *module)
{
  static const GTypeInfo our_info =
  {
    sizeof (EphyFooExtensionClass),
    NULL, /* base_init */
    NULL, /* base_finalize */
    (GClassInitFunc) ephy_foo_extension_class_init,
    NULL,
    NULL, /* class_data */
    sizeof (EphyFooExtension),
    0, /* n_preallocs */
    (GInstanceInitFunc) ephy_foo_extension_init
  };

  static const GInterfaceInfo extension_info =
  {
    (GInterfaceInitFunc) ephy_foo_extension_iface_init,
    NULL,
    NULL
  };

  foo_extension_type =
    g_type_module_register_type (module, G_TYPE_OBJECT, "EphyFooExtension", &our_info, 0);
  
  g_type_module_add_interface (module, foo_extension_type, EPHY_TYPE_EXTENSION, &extension_info);

  return foo_extension_type;
}

static void
ephy_foo_extension_init (EphyFooExtension *extension)
{
  extension->priv = EPHY_FOO_EXTENSION_GET_PRIVATE (extension);

  LOG ("EphyFooExtension initialising")
}

static void
ephy_foo_extension_finalize (GObject *object)
{
  /* If needed,
   * EphyFooExtension *extension = EPHY_FOO_EXTENSION (object);
   * ...followed by, say, some g_free calls.
   */
  
  LOG ("EphyFooExtension finalizing")

  G_OBJECT_CLASS (foo_extension_parent_class)->finalize (object);
}

static void
ephy_foo_extension_set_property (GObject *object, guint prop_id, const GValue *value, GParamSpec *pspec)
{
/* Code this function if anything was placed after PROP_0 in the enum above. */
/*
  EphyFooExtension *extension = EPHY_FOO_EXTENSION (object);

  switch (prop_id)
  {
  }
*/
}

static void
ephy_foo_extension_get_property (GObject *object, guint prop_id, GValue *value, GParamSpec *pspec)
{
/* Code this function if anything was placed after PROP_0 in the enum above. */
/*
  EphyFooExtension *extension = EPHY_FOO_EXTENSION (object);

  switch (prop_id)
  {
  }
*/
}

static void
ephy_foo_extension_class_init (EphyFooExtensionClass *klass)
{
  GObjectClass *object_class = G_OBJECT_CLASS (klass);

  foo_extension_parent_class = g_type_class_peek_parent (klass);

  object_class->finalize = ephy_foo_extension_finalize;
  object_class->get_property = ephy_foo_extension_get_property;
  object_class->set_property = ephy_foo_extension_set_property;

  g_type_class_add_private (object_class, sizeof (EphyFooExtensionPrivate));
}
