4.2. Header File

The foo.c file above referred to ephy-foo-extension.h, so the header file must be written. Also, as hinted above, an extension is a GObject. There are no surprises in the header file: just standard GObject stuff:

/* header information */
#ifndef EPHY_FOO_EXTENSION_H
#define EPHY_FOO_EXTENSION_H

#include <glib.h>
#include <glib-object.h>

G_BEGIN_DECLS

#define EPHY_TYPE_FOO_EXTENSION (ephy_foo_extension_get_type ())
#define EPHY_FOO_EXTENSION(o) (G_TYPE_CHECK_INSTANCE_CAST ((o, EPHY_TYPE_FOO_EXTENSION, EphyFooExtension))
#define EPHY_FOO_EXTENSION_CLASS(k) (G_TYPE_CHECK_CLASS_CAST((k), EPHY_TYPE_FOO_EXTENSION, EphyFooExtensionClass))
#define EPHY_IS_FOO_EXTENSION(o) (G_TYPE_CHECK_INSTANCE_TYPE ((o), EPHY_TYPE_FOO_EXTENSION))
#define EPHY_IS_FOO_EXTENSION_CLASS(k) (G_TYPE_CHECK_CLASS_TYPE ((k), EPHY_TYPE_FOO_EXTENSION))
#define EPHY_FOO_EXTENSION_GET_CLASS(o) (G_TYPE_INSTANCE_GET_CLASS ((o), EPHY_TYPE_FOO_EXTENSION, EphyFooExtensionClass))

typedef struct EphyFooExtension EphyFooExtension;
typedef struct EphyFooExtensionClass EphyFooExtensionClass;
typedef struct EphyFooExtensionPrivate EphyFooExtensionPrivate;

struct EphyFooExtensionClass
{
	GObjectClass parent_class;
};

struct EphyFooExtension
{
	GObject parent_instance;

	EphyFooExtensionPrivate *priv;
};

GType ephy_foo_extension_get_type (void);
GType ephy_foo_extension_register_type (GTypeModule *module);

G_END_DECLS

#endif
